Calculating selected prevalences from ihme dataset

## Get cause_id what we can use to overwrite base values
tbl(con, "paf") %>% 
  select(Cause_ID) %>% 
  collect() -> causes
causes <- causes$Cause_ID[!is.na(causes$Cause_ID)]

## Get number of prevalences per location and causes from IHME
tbl(con, "ihme") %>% 
  filter(cause_id %in% causes & metric_name == "Number") %>% 
  group_by(location_name, cause_id, cause_name) %>% 
  summarise(prevalence_number = sum(val)) %>% 
  collect() -> prevalences_ihme
## `summarise()` has grouped output by 'location_name', 'cause_id'. You can override using the `.groups` argument.

Calculating 15-74 years old population of the countries

## Only total population
tbl(con, "pop") %>% 
  filter(sex_name == "both" & age_group > 14 & age_group < 75 & year_id == 2019) %>% 
  group_by(location_name) %>% 
  summarise(pop = sum(total_population)) %>% 
  collect() -> population_age_filtered
## Also female and male population in wide format
library(tidyr)
tbl(con, "pop") %>% 
  filter(age_group > 14 & age_group < 75 & year_id == 2019) %>% 
  group_by(location_name, sex_name) %>% 
  summarise(pop = sum(total_population)) %>% 
  collect() %>% 
  pivot_wider(names_from = sex_name, values_from = pop) -> population_age_filtered
## `summarise()` has grouped output by 'location_name'. You can override using the `.groups` argument.

Calculating prevalences

## join tables
prevalences_ihme %>% 
  right_join(population_age_filtered, by = "location_name") %>% 
  mutate(prevalence = prevalence_number / both) -> prevalences

Combining prevalences with base values

## Base data. We need paf info for every country
d <- NULL
for(cntry in unique(prevalences$location_name)){
  # print(cntry)
  tbl(con, "paf") %>% 
    collect() %>% 
    mutate(location_name = cntry,
           condition = Conditon) %>% 
    rename(cause_id=Cause_ID) %>% 
    select("location_name", 
           "cause_id", 
           "condition", 
           "PAF", 
           "prevalence", 
           "direct_healtcare_cost_mean", 
           "direct_nonhealthcare_cost_mean", 
           "productivity_losses_cost_mean") %>% 
    rbind(d) -> d
}

## Filter prevalence table to only what we need
prevalences <- prevalences %>% 
  select(location_name, cause_id, cause_name, prevalence, both, female, male)

## Join base data and new calculated prevalences
d <- d %>% 
  left_join(prevalences, by = c("location_name", "cause_id")) %>% 
  mutate(prevalence = ifelse(is.na(prevalence.y), prevalence.x, prevalence.y)) %>% 
  rename(prevalence_base = prevalence.x, 
         prevalence_ihme = prevalence.y) %>% 
  select(location_name, cause_id, cause_name, condition, PAF, prevalence_base, prevalence_ihme, prevalence, direct_healtcare_cost_mean, direct_nonhealthcare_cost_mean, productivity_losses_cost_mean)

## Add the population
d %>% 
  right_join(population_age_filtered, by = "location_name") %>% 
  rename(
         population_total = both,
         population_female = female,
         population_male = male) -> d
dbDisconnect(con, shutdown=TRUE)
