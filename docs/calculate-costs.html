6.1 Per Country

Formulas to total costs are

tbl(con, "prevalences") %>% 
  mutate(
    ## Formulas
    prevalent_cases = prevalence * population_total,
    prevalent_cases_influenced_osa = PAF * prevalent_cases,
    direct_cost = prevalent_cases_influenced_osa * direct_healtcare_cost_mean,
    direct_non_healthcare_cost = prevalent_cases_influenced_osa * direct_nonhealthcare_cost_mean,
    productivity_lost_cost = prevalent_cases_influenced_osa * productivity_losses_cost_mean,
    total_costs = direct_cost + direct_non_healthcare_cost + productivity_lost_cost
  ) %>% 
  collect() -> slapnea_costs

Formulas for per patient costs are

## Calculate sums of direct, non healthcare and productivity lost costs.
slapnea_costs %>% 
  group_by(location_name, population_female, population_male) %>% 
  summarise(direct_cost = sum(direct_cost, na.rm = T),
            direct_non_healthcare_cost = sum(direct_non_healthcare_cost, na.rm = T),
            productivity_lost_cost = sum(productivity_lost_cost, na.rm = T)) -> summary_slapnea_costs
## `summarise()` has grouped output by 'location_name', 'population_female'. You can override using the `.groups` argument.
## Calculate per patient cost
## by calculating absolute values (population) using OSA table (slapnea)
summary_slapnea_costs %>% 
  mutate(
    ## Formulas
    ## TODO: tarkista 
    absolute_value_severe_moderate = ( (population_female * slapnea$rate[slapnea$var == "Moderate" & slapnea$gender == "Female"]) + (population_female * slapnea$rate[slapnea$var == "Severe" & slapnea$gender == "Female"]) + (population_male * slapnea$rate[slapnea$var == "Moderate" & slapnea$gender == "Male"]) + (population_male * slapnea$rate[slapnea$var == "Severe" & slapnea$gender == "Male"])),
    absolute_value_mild = ( population_female * slapnea$rate[slapnea$var == "Mild" & slapnea$gender == "Female"] + population_male * slapnea$rate[slapnea$var == "Mild" & slapnea$gender == "Male"] ),
    patient_direct_cost = direct_cost / absolute_value_severe_moderate,
    patient_nondirect_cost = direct_non_healthcare_cost / absolute_value_mild,
    patient_productivity_cost = productivity_lost_cost / absolute_value_mild,
    patient_total_cost = patient_direct_cost + patient_nondirect_cost + patient_productivity_cost
  ) -> summary_slapnea_costs
dbDisconnect(con, shutdown=TRUE)

6.4 Save dataset

Saving dataset to new duckdb database and creating parquet files for longer storage.

library(duckdb)
con = dbConnect(duckdb::duckdb(), dbdir="./data/sleep22_calculator.duckdb", read_only=FALSE)
dbWriteTable(con, "prevalences", d, overwrite=TRUE)
dbWriteTable(con, "summary_slapnea_costs", summary_slapnea_costs, overwrite=TRUE)
dbWriteTable(con, "slapnea_costs", slapnea_costs, overwrite=TRUE)

dbDisconnect(con)
